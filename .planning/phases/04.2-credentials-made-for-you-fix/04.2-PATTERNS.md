```rust
/// Acquire reusable credentials from a Spotify OAuth access token.
/// Calls session.connect() with store_credentials=true, which writes
/// credentials.json to cache_dir via the automatic token->reusable_credentials
/// exchange in librespot-core (session.rs:184-201).
async fn run_token_login(
    access_token: &str,
    cache_dir: &str,
) -> Result<(), Box<dyn std::error::Error>> {
    // Credentials::with_access_token() setzt auth_type = AUTHENTICATION_SPOTIFY_TOKEN
    // (verifiziert: librespot-core-0.8.0/src/authentication.rs:61-67)
    let credentials = Credentials::with_access_token(access_token);

    // Cache-Pfad analog run_authenticate() Zeile 282
    let cache = Cache::new(Some(cache_dir), None::<&str>, None::<&str>, None)?;

    let session_config = SessionConfig::default();
    let session = Session::new(session_config, Some(cache));

    // store_credentials=true: librespot-core verbindet intern zweimal
    // (Token -> AP -> reusable_credentials -> credentials.json in cache_dir)
    // (verifiziert: session.rs:184-201, session.rs:249-258)
    session.connect(credentials, true).await?;

    Ok(())
}
```

**Core-Pattern fuer `_provisionCredentials`** (analog zu helperCheck-Muster in Helper.pm):
```perl
# _provisionCredentials($class, $accountId, $accessToken)
# Ruft den Binary im --token-login Modus auf, um credentials.json in den
# SpotOn-Cache-Dir zu schreiben. Blockierender system()-Aufruf ist hier
# akzeptabel: laeuft einmalig nach OAuth-Callback, Laufzeit < 2s (A3).
# SECURITY (T-02.1-03): $accessToken darf NIEMALS geloggt werden.
sub _provisionCredentials {
    my ($class, $accountId, $accessToken) = @_;

    require Plugins::SpotOn::Helper;
    my ($helper) = Plugins::SpotOn::Helper->get();
    unless ($helper) {
        $log->warn("TokenManager: cannot provision credentials — binary not found");
        return;
    }

    # Cache-Dir: analog Plugin.pm::updateTranscodingTable Zeile 1108-1109
    my $serverPrefs = preferences('server');
    my $cacheDir    = catdir($serverPrefs->get('cachedir'), 'spoton');

    # Shell-Escaping fuer einfache Anführungszeichen
    (my $safeHelper = $helper)      =~ s/'/'\\''/g;
    (my $safeToken  = $accessToken) =~ s/'/'\\''/g;
    (my $safeDir    = $cacheDir)    =~ s/'/'\\''/g;

    my $cmd = sprintf("'%s' -n 'SpotOn' --token-login --token '%s' --cache '%s' 2>&1",
        $safeHelper, $safeToken, $safeDir);

    # T-02.1-03: Log-Zeile verwendet [REDACTED], nie den Token selbst
    main::INFOLOG && $log->info(
        "TokenManager: provisioning credentials for account $accountId " .
        "(cmd: '$safeHelper' -n 'SpotOn' --token-login --token [REDACTED] --cache '$safeDir')");

    my $output = `$cmd`;
    my $exit   = $? >> 8;

    if ($exit == 0 && ($output =~ /credentials_saved/)) {
        main::INFOLOG && $log->info(
            "TokenManager: credentials provisioned for account $accountId");
    } else {
        $log->error(
            "TokenManager: credential provisioning failed for $accountId (exit $exit): $output");
        # Kein hard failure — naechster --single-track-Aufruf meldet "No cached credentials"
    }
}
```

**Error-Handling-Pattern** (analog zu `refreshToken` Zeilen 118-128):

#### 2. `_madeForYouFeed()` — Category-Endpoint + Fallback (Zeilen 482-497)

```perl
# _madeForYouFeed($client, $callback, $args)
# Primary: Category endpoint (D-05) — GET browse/categories/{id}/playlists
# Fallback: me/playlists + _isMadeForYou name-matching (D-06) when 404/403 (Dev Mode)
# Response structure: $data->{playlists}{items} — NOT $data->{items} (Pitfall 2)
sub _madeForYouFeed {
    my ($client, $callback, $args) = @_;
    my $accountId = _getAccountId($client);

    Plugins::SpotOn::API::Client->getPersonalMixes($accountId, {}, sub {
        my ($data, $err) = @_;

        # Fallback bei 404/403 (Dev Mode: Browse Categories seit Feb 2026 entfernt)

            main::INFOLOG && $log->info(
                "_madeForYouFeed: category endpoint unavailable (code " .
                ($err ? $err->{code} : 'no data') . "), falling back to me/playlists");

                { offset => 0, limit => 50 }, sub {

                $callback->({ items => \@items });

                    @{ $data->{playlists}{items} || [] };

    });
```
